# ReactBricks

The `<ReactBricks>` component wraps all the other React Bricks components and acts as a Provider for the [`<ReactBricksContext>`](/api-reference/hooks/use-react-bricks-context) that make the [configuration](/docs/configuration/configuration) available to all the other components.

## Usage example

```tsx
<ReactBricks
  appId="XXXXXXXX-XXXX-XXXX-XXXX-XXXXXXXXXXXX"
  apiKey="XXXXXXXX-XXXX-XXXX-XXXX-XXXXXXXXXXXX"
  bricks={...}
  pageTypes={...}
  logo="/logo.svg"
  contentClassName="content"
  renderLocalLink={({ href, children, className, activeClassName }) => {...}}
  navigate={...}
  loginPath="/"
  editorPath="/editor"
  playgroundPath="/playground"
  appSettingsPath="/app-settings"
  isDarkColorMode={colorMode === 'dark'}
  toggleColorMode={...}
  appRootElement="#root"
  clickToEditSide={types.ClickToEditSide.BottomRight}
  customFields={...}
  responsiveBreakpoints={...}
  enableAutoSave
  disableSaveIfInvalidProps
>
  ...
</ReactBricks>
```
